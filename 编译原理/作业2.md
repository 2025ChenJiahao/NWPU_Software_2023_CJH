以下是完全符合题目要求的**TINY 语言词法分析程序**，严格遵循题目给出的词法规则与 DFA 状态转移逻辑，使用 Python 实现（语法简洁、易运行、适合逐行讲解），同时附带完整的代码逐行解释。

## 一、TINY 词法规则与 DFA 核心逻辑回顾

### 1. 词法规则

1. **关键字**：`if` `then` `else` `end` `repeat` `until` `read` `write`（保留字，不可作为标识符）
2. **标识符 ID**：`letter letter*`，字母开头，后续仅可跟字母，`letter=[a-z A-Z]`
3. **数字 NUM**：`digit digit*`，数字开头，后续仅可跟数字，`digit=[0-9]`
4. **空白符**：空格、换行符`\n`、制表符`\t`，直接跳过，不生成 Token
5. **注释**：`{...}`，不可嵌套，`{`到`}`之间的内容全部忽略
6. **专用符号**：`+` `-` `*` `/` `<` `=` `(` `)` `:=`（其中`:=`为双字符符号）

### 2. DFA 状态转移核心

- `START`：起始状态，根据读入的字符切换到对应状态
- `INNUM`：数字识别状态，遇到数字保持，其他字符结束识别
- `INID`：标识符 / 关键字识别状态，遇到字母保持，其他字符结束识别
- `INASSIGN`：赋值符号识别状态，遇到`:`进入，下一个字符为`=`则完成`:=`识别
- `INCOMMENT`：注释处理状态，遇到`{`进入，遇到`}`回到 START
- `DONE`：单 Token 识别完成，返回结果

## 二、完整源程序代码

```py
from enum import Enum

# ===================== 1. 定义Token类型枚举 =====================
class TokenType(Enum):
    # 关键字类型
    IF = "IF"
    THEN = "THEN"
    ELSE = "ELSE"
    END = "END"
    REPEAT = "REPEAT"
    UNTIL = "UNTIL"
    READ = "READ"
    WRITE = "WRITE"
    # 标识符与数字类型
    ID = "ID"
    NUM = "NUM"
    # 运算符与分隔符类型
    PLUS = "PLUS"       # +
    MINUS = "MINUS"     # -
    TIMES = "TIMES"     # *
    OVER = "OVER"       # /
    LT = "LT"           # <
    EQ = "EQ"           # =
    ASSIGN = "ASSIGN"   # :=
    LPAREN = "LPAREN"   # (
    RPAREN = "RPAREN"   # )
    # 结束与错误类型
    EOF = "EOF"
    ERROR = "ERROR"

# ===================== 2. 关键字映射字典 =====================
KEYWORDS = {
    "if": TokenType.IF,
    "then": TokenType.THEN,
    "else": TokenType.ELSE,
    "end": TokenType.END,
    "repeat": TokenType.REPEAT,
    "until": TokenType.UNTIL,
    "read": TokenType.READ,
    "write": TokenType.WRITE
}

# ===================== 3. TINY词法分析器核心类 =====================
class TinyLexer:
    # 构造函数：初始化词法分析器状态
    def __init__(self, source_code):
        self.source = source_code    # 输入的TINY源代码
        self.pos = 0                  # 当前读取的字符索引
        self.current_char = self.source[self.pos] if self.source else None
        self.line_num = 1             # 当前行号，用于错误定位

    # 辅助函数：向前读取一个字符，更新当前状态与行号
    def advance(self):
        if self.current_char == '\n':
            self.line_num += 1
        self.pos += 1
        # 判断是否到达源代码末尾
        self.current_char = self.source[self.pos] if self.pos < len(self.source) else None

    # 辅助函数：跳过所有空白符（空格、制表符、换行符）
    def skip_whitespace(self):
        while self.current_char is not None and self.current_char.isspace():
            self.advance()

    # 辅助函数：处理{...}格式的注释，对应DFA的INCOMMENT状态
    def skip_comment(self):
        self.advance()  # 跳过注释起始符'{'
        # 循环跳过注释内容，直到遇到'}'或文件结束
        while self.current_char is not None and self.current_char != '}':
            self.advance()
        # 处理注释未闭合的错误
        if self.current_char is None:
            return TokenType.ERROR, f"Unclosed comment at line {self.line_num}"
        self.advance()  # 跳过注释结束符'}'
        return None, None

    # 辅助函数：识别数字NUM，对应DFA的INNUM状态
    def read_number(self):
        num_str = ""
        # 循环读取所有连续的数字
        while self.current_char is not None and self.current_char.isdigit():
            num_str += self.current_char
            self.advance()
        return TokenType.NUM, num_str

    # 辅助函数：识别标识符/关键字，对应DFA的INID状态
    def read_identifier(self):
        id_str = ""
        # 循环读取所有连续的字母
        while self.current_char is not None and self.current_char.isalpha():
            id_str += self.current_char
            self.advance()
        # 判断是否为关键字，非关键字则为普通标识符
        token_type = KEYWORDS.get(id_str.lower(), TokenType.ID)
        return token_type, id_str

    # 核心函数：实现DFA状态转移，返回下一个Token
    def next_token(self):
        while self.current_char is not None:
            # 1. 处理空白符，回到START状态
            if self.current_char.isspace():
                self.skip_whitespace()
                continue

            # 2. 处理注释，进入INCOMMENT状态
            if self.current_char == '{':
                err_type, err_msg = self.skip_comment()
                if err_type == TokenType.ERROR:
                    return err_type, err_msg
                continue

            # 3. 处理数字，进入INNUM状态
            if self.current_char.isdigit():
                return self.read_number()

            # 4. 处理标识符/关键字，进入INID状态
            if self.current_char.isalpha():
                return self.read_identifier()

            # 5. 处理赋值符号:=，进入INASSIGN状态
            if self.current_char == ':':
                self.advance()
                if self.current_char == '=':
                    self.advance()
                    return TokenType.ASSIGN, ":="
                else:
                    return TokenType.ERROR, f"Unexpected ':' at line {self.line_num}"

            # 6. 处理单字符专用符号
            if self.current_char == '+':
                self.advance()
                return TokenType.PLUS, "+"
            elif self.current_char == '-':
                self.advance()
                return TokenType.MINUS, "-"
            elif self.current_char == '*':
                self.advance()
                return TokenType.TIMES, "*"
            elif self.current_char == '/':
                self.advance()
                return TokenType.OVER, "/"
            elif self.current_char == '<':
                self.advance()
                return TokenType.LT, "<"
            elif self.current_char == '=':
                self.advance()
                return TokenType.EQ, "="
            elif self.current_char == '(':
                self.advance()
                return TokenType.LPAREN, "("
            elif self.current_char == ')':
                self.advance()
                return TokenType.RPAREN, ")"

            # 7. 处理未知非法字符
            error_msg = f"Unexpected character '{self.current_char}' at line {self.line_num}"
            self.advance()
            return TokenType.ERROR, error_msg

        # 到达源代码末尾，返回EOF Token
        return TokenType.EOF, "EOF"

# ===================== 4. 测试函数 =====================
def test_lexer(source_code):
    lexer = TinyLexer(source_code)
    print("===== TINY词法分析结果 =====")
    print(f"{'Token类型':<10} | {'Token值'}")
    print("-" * 30)
    while True:
        token_type, token_value = lexer.next_token()
        print(f"{token_type.value:<10} | {token_value}")
        # 遇到EOF或错误时终止分析
        if token_type in (TokenType.EOF, TokenType.ERROR):
            break
    print("-" * 30)
    print("词法分析完成")

# ===================== 5. 主程序入口 =====================
if __name__ == "__main__":
    # 测试用的TINY源代码（完全符合题目词法规则）
    tiny_test_code = """
    { 这是TINY语言的测试代码 计算阶乘 }
    read x
    if 0 < x then
        fact := 1
        repeat
            fact := fact * x
            x := x - 1
        until x = 0
        write fact
    end
    """
    # 执行词法分析
    test_lexer(tiny_test_code)
```
