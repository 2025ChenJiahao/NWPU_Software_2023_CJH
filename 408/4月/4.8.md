## 兄弟节点：填充每个节点指向兄弟的指针：所有情况，不管你是不是完全二叉反正都一样，就是慢

```py
class Solution:
    def connect(self, root: 'Optional[Node]') -> 'Optional[Node]':
        if not root:
            return None

        from collections import deque
        q = deque([root])
        res=[]
        while q:
            level_size = len(q)
            for i in range(level_size):

                node = q.popleft()
                res.append(node.val)
                # 同一个层里，不是最后一个就指向右边的节点
                if i < level_size - 1:
                    node.next = q[0]

                else:
                    node.next = None
                    res.append("#")
                # 打印当前节点 next 指向
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
        return root  #res
```

## 二叉树找最大路径和（关键是二叉树，不然要需要向上考虑）

```py
class Solution:
    def maxPathSum(self, root: Optional[TreeNode]) -> int:
        self.max_sum = float('-inf')
        def dfs(node):
            if not node:
                return 0
            # 左、右子树能提供的最大增益，负数就不要
            left = max(dfs(node.left), 0)
            right = max(dfs(node.right), 0)
            # 以当前节点为顶点的完整路径和
            current = node.val + left + right
            if current > self.max_sum:
                self.max_sum = current
            # 返回给父节点：只能选一条链
            return node.val + max(left, right)
        dfs(root)
        return self.max_sum
```

## 求根节点到叶子节点所有路径上的val的数字之和（1->2->3===123）

```py
class Solution:
    def sumNumbers(self, root: Optional[TreeNode]) -> int:
        #我们知道每个叶子节点代表了一个数字，所以说向下循环即可哈哈哈哈num=num*10+current
        res=[]
        def dfs(node, current_num):
            if not node:
                return
            # 核心公式：你说的没错！
            current_num = current_num * 10 + node.val
            # ✅ 到达叶子节点：把路径数字加入结果
            if not node.left and not node.right:
                res.append(current_num)#这是核心好吧
                return
            # ✅ 递归左右
            dfs(node.left, current_num)
            dfs(node.right, current_num)
        dfs(root, 0)
        return sum(res)  # 最后求和
```

```py
def main():
    data = list(map(int, sys.stdin.read().split()))
    idx = 0
    N = data[idx]
    idx += 1
    K = data[idx]
    idx += 1
    hw = []
    max_side = 0
    for _ in range(N):
        h = data[idx]
        idx += 1
        w = data[idx]
        idx += 1
        hw.append((h, w))
        max_side = max(max_side, h, w)
    # 二分答案
    left = 1
    right = max_side
    ans = 0
    while left <= right:
        mid = (left + right) // 2
        total = 0
        for h, w in hw:
            total += (h // mid) * (w // mid)
            if total >= K:  # 提前剪枝
                break
        if total >= K:
            ans = mid      # 记录答案
            left = mid + 1 # 尝试更大
        else:
            right = mid - 1
    print(ans)
main()
```

```py
import sys

# 读取输入
start = input().strip()
end = input().strip()

# BFS 队列：(当前状态, 空杯位置, 步数)
q = deque()
# 记录访问过的状态，避免重复走
visited = set()

# 找到初始 * 的位置
star_pos = start.index('*')
q.append((start, star_pos, 0))
visited.add(start)

# 青蛙可以跳的方向：-3,-2,-1,1,2,3（跳1、2、3格）
dirs = [-3, -2, -1, 1, 2, 3]

# 开始 BFS
while q:
    state, pos, step = q.popleft()

    # 到达目标，直接输出步数
    if state == end:
        print(step)
        sys.exit()

    # 转成列表方便交换（字符串不能直接改）
    s = list(state)

    # 尝试所有 6 种跳法
    for d in dirs:
        new_pos = pos + d  # 跳到的新位置
        if 0 <= new_pos < len(s):
            # 交换空杯和青蛙的位置
            s[pos], s[new_pos] = s[new_pos], s[pos]
            new_state = ''.join(s)

            # 没走过就加入队列
            if new_state not in visited:
                visited.add(new_state)
                q.append((new_state, new_pos, step + 1))

            # 恢复原状，继续尝试下一种跳法
            s[pos], s[new_pos] = s[new_pos], s[pos]
```

```py
def delete_until_end(s):

        changed = False

            c = s[i + 2]

                changed = True
```

## 重新整理数组后最大和

$$
给定一个数组 A 和一些查询 Li,Ri, 求数组中第 Li 至第 Ri个元素之和。
$$

小蓝觉得这个问题很无聊, 于是他想重新排列一下数组, 使得最终每个查 询结果的和尽可能地大。小蓝想知道相比原数组, 所有查询结果的总和最多可 以增加多少?

```py
import sys


def solve():
    # 1. 读取数据
    all_data = sys.stdin.read().split()
    if not all_data: return

    n = int(all_data[0])
    arr = list(map(int, all_data[1:n + 1]))
    m = int(all_data[n + 1])

    queries = []
    start_pos = n + 2
    for i in range(start_pos, start_pos + 2 * m, 2):
        queries.append((int(all_data[i]), int(all_data[i + 1])))

    # 2. 计算原始区间和之和
    # 先构造原数组的前缀和
    prefix_sum = [0] * (n + 1)
    for i in range(n):
        prefix_sum[i + 1] = prefix_sum[i] + arr[i]

    original_total = 0
    for l, r in queries:
        # 区间 [l, r] 的和公式：S[r] - S[l-1]
        original_total += (prefix_sum[r] - prefix_sum[l - 1])

    # 3. 统计每个位置被查询的次数 (使用差分数组)
    count_diff = [0] * (n + 2)
    for l, r in queries:
        count_diff[l] += 1
        count_diff[r + 1] -= 1

    # 还原差分数组，得到每个位置实际的查询次数
    counts = [0] * (n + 1)
    curr = 0
    for i in range(1, n + 1):
        curr += count_diff[i]
        counts[i] = curr

    # 4. 贪心重排
    # 过滤掉不需要统计的 0 下标，只取 1~n 的频率并排序
    freq = sorted(counts[1:], reverse=True)
    # 将原数组也从大到小排序
    sorted_arr = sorted(arr, reverse=True)

    # 计算重排后的最大和
    new_total = 0
    for i in range(n):
        # 最大的数 * 最高的频率
        new_total += sorted_arr[i] * freq[i]

    # 5. 输出增加的值
    print(new_total - original_total)


solve()
```

```py
def solve():
    import sys
    input = sys.stdin.read
    data = input().split()

    if not data:
        return

    n = int(data[0])
    MOD = 998244353

    if n < 2:
        print(0)
        return

    # 核心公式: Ans = (n! * n * (n-1) / 4) % MOD

    # 1. 计算 n! % MOD
    fact = 1
    for i in range(1, n + 1):
        fact = (fact * i) % MOD

    # 2. 计算 n * (n-1) / 4
    # 这里分情况处理除法，避免浮点数，或者直接用 pow(4, MOD-2, MOD) 求逆元
    # 为了通用性，使用费马小定理求 4 的逆元
    inv4 = pow(4, MOD - 2, MOD)

    # 3. 组合计算
    term2 = (n * (n - 1)) % MOD
    ans = (fact * term2) % MOD
    ans = (ans * inv4) % MOD

    print(ans)
```

```py
num = int(input())

def prime_factor_count(n):
    """
    输入正整数 n, 返回 n 的质因数个数。
    """
    res = dict()
    # 先将质因数2，3 除去；
    for i in [2, 3]:
        while n != i:
            if n % i == 0:
                if i in res: res[i] += 1  # 记录质因数：质因数出现次数；
                else:        res[i] = 1
                n //= i
            else: break

    # 再将6n+1，6n+5除去；
    for j in range(5, int(n**0.5)+1, 6):
        for i in [j, j+2]:  # j:6n+5; j+2: 6(n+1)+1
            while n != i:
                if n % i == 0:
                    if i in res: res[i] += 1
                    else:        res[i] = 1
                    n //= i
                else: break

    if n > 0: res[n] = 1 # n最后不为0，说明n也是一个质数；

    # print(res)
    return len(res)
```
