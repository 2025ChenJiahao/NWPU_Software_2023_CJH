            l = r
        else:
            l += 1
    smax = (smax + add * pow(65535, sum(bits), mod)) % mod

## 三、路径规划简单的剪枝

```py
import os
import sys
import copy

sys.setrecursionlimit(1500000) # 因为要走很多很多步，怕电脑报错说“你想得太深了”，所以提高电脑的思考深度上限。
n = int(input()) # 地图是 n*n 的大小
vis = [[0] * n for _ in range(n)] # 你的“脚印记号本”。0代表没踩过，1代表踩过。
lie = list(map(int, input().split())) # 每【列】需要踩的次数清单
hang = list(map(int, input().split())) # 每【行】需要踩的次数清单
tar = (sum(hang) + sum(lie)) // 2  # 一共要走多少步？（因为踩一个格子既算一行又算一列，所以总和除以2就是总步数）
dirs = [[0, 1], [1, 0], [-1, 0], [0, -1]] # 你的指南针：分别代表向右、向下、向上、向左走一步。
p = "" # 用来记录最终正确路线的小本本。


def dfs(x, y, path, bu):#x, y：你现在站在哪行哪列。path：你一路走过来记下的坐标路线。bu：你现在走了第几步。
    global p
    vis[x][y] = 1  # 在脚印记号本上写：这里我走过了！
    hang[x] -= 1  # 这一行的任务划掉一个（还要踩的次数减1）
    lie[y] -= 1  # 这一列的任务划掉一个
    if hang[x] == 0:#如果你发现即使继续走，也肯定完不成任务了，不如直接放弃这条路。这就叫“剪枝”。
        for i in range(x):#这是什么意思？ 意思是如果你把当前这一行的任务做完了（hang[x] == 0），
            if hang[i] > 0:#你就回头看看上面的行（range(x)）。如果上面的行竟然还有任务没做完（hang[i] > 0），那你完蛋了！
                return#因为按照规则你很难再绕回上面去补救。既然这条路注定失败，直接 return（放弃这条路，时光倒流）。对列（lie）的检查也是同理。
    if lie[y] == 0:
        for i in range(y):
            if lie[i] > 0:
                return
    if x == n - 1 and y == n - 1:  # 如果你走到右下角了
        if sum(hang) == 0 and sum(lie) == 0 and bu == tar:  # 检查所有行和列的任务是不是都刚好为0，且步数正好？
            p = path  # 太棒了！把这条正确的路线记到总本子 p 上！
            return
        else:
            return  # 虽然到了终点，但任务没完成，这也是一条废路，放弃。
    for xq, yq in dirs:  # 拿出指南针，尝试四个方向
        xi, yi = x + xq, y + yq  # 算出下一步要踩的坐标 (xi, yi)

        # 如果下一步没出界，没踩过，而且那个格子所在的行和列都还有任务额度 (hang > 0, lie > 0)
        if 0 <= xi < n and 0 <= yi < n and vis[xi][yi] != 1 and hang[xi] > 0 and lie[yi] > 0:
            # 【动作A：往前走】
            # xi * n + yi 是把你 (行,列) 的二维坐标换算成 0,1,2,3... 这样的一个编号。
            dfs(xi, yi, path + "/" + str(xi * n + yi), bu + 1)

            # 【动作B：时光倒流（拔旗 / 回溯）】
            vis[xi][yi] = 0
            hang[xi] += 1
            lie[yi] += 1


dfs(0, 0, "0", 1) # 游戏开始！你站在 (0,0)，当前路线记为 "0"，这是第 1 步。

# 游戏结束后，p 里面装的就是正确的路径，类似 "0/1/5/9" 这样。
for i in p.split("/"): # 把路线按 "/" 拆开，打印出来
    print(i,end=" ")
```

## 四、拼接二进制数使其最大

```py
import sys
#sys.setrecursionlimit(1000000)
# sys.set_int_max_str_digits(1000000)
from functools import cmp_to_key
n = int(input())
nums = [bin(i)[2:] for i in range(1, n + 1)]
nums.sort(key=cmp_to_key(lambda a, b: 1 if a + b < b + a else -1))
print(int(''.join(nums), 2))
```

### 1. 核心原理：什么是比较函数 (cmp)？

一个传统的比较函数接收两个参数 `(x, y)`，并返回一个整数：

- **返回负数**（如 `-1`）：表示 $x < y$（$x$ 应该排在 $y$ 前面）。
- **返回正数**（如 `1`）：表示 $x > y$（$x$ 应该排在 $y$ 后面）。
- **返回 0**：表示两者相等。

------

### 2. 常见使用场景

#### 场景 A：自定义组合逻辑（如：最大数问题）

这是最经典的使用场景。假设你有两个数字 `9` 和 `91`，你想把它们拼成最大的数。

- 如果按默认排序：`91, 9` --------->拼接成 `919`
- 如果按“自定义逻辑”排序：比较 `9+91` 和 `91+9`，发现 `991 > 919`，所以 `9` 应该排在 `91` 前面。

```
from functools import cmp_to_key

def compare_logic(a, b):
    # 尝试两种拼接方式
    if a + b > b + a:
        return -1  # a 应该在前面
    else:
        return 1   # b 应该在前面

nums = ["9", "91", "2"]
# 必须转换，因为 key 不支持传入两个参数
sorted_nums = sorted(nums, key=cmp_to_key(compare_logic))
print("".join(sorted_nums))  # 输出: 9912
```

#### 场景 B：多级或复杂的对象排序

当你需要根据某些外部规则（比如一份复杂的映射表）来决定两个对象谁先谁后，且这种规则很难用简单的 `key=lambda x: ...` 表达时。

#### 场景 C：兼容旧代码

如果你正在从 Python 2 迁移到 Python 3，以前的代码里写满了自定义的 `cmp` 函数，使用 `cmp_to_key` 可以让你在不重写核心算法逻辑的情况下，快速适配 Python 3。

------

### 3. 为什么不直接用 key？

大多数时候，`key` 性能更好。例如，按绝对值排序：

- **Key 方式**：`sorted(list, key=abs)`。每个元素只调用一次 `abs()`，然后对比结果。
- **Cmp 方式**：每次两两比较都要重新计算逻辑，效率较低。

**结论**：

- 如果排序顺序能通过**观察单个元素**得出（如：长度、数值、某个字段），用 **`key`**。
- 如果排序顺序必须**通过两个元素互相博弈**得出（如：A+B 和 B+A 谁大），用 **`cmp_to_key`**。

------

由于网格图的拉普拉斯矩阵是一个特殊的**带状矩阵（Banded Matrix）**（宽度仅为 $n=6$），代码通过高斯消元法利用这一特性，在 $O(S \cdot N^2)$ 的时间复杂度内完成了行列式计算（其中 $S$ 是顶点数，$N$ 是网格高度）。

```
# 定义模数，通常用于防止数值溢出并满足题目要求
mod = int(1e9) + 7

# 快速幂函数：计算 (a^b) % mod
# 原理：利用二进制拆分指数，将复杂度降为 O(log b)
def fpow(a, b):
    res = 1
    while b:
        if b & 1: # 如果 b 的当前二进制位为 1
            res = (res * a) % mod
        a = (a * a) % mod # 基数平方
        b >>= 1 # 指数右移一位
    return res

# 读取网格的行数 n 和列数 m
n, m = map(int, input().split())
# 读取地图，'E' 可能代表空地（节点），其他代表障碍
buffer = [input() for _ in range(n)]

# p 数组用于给网格中的每个 'E' 节点进行唯一编号（从 1 开始）
# p[行][列] = 该点的 ID
p = [[0] * 100005 for _ in range(6)]
s = 0 # 计数器，记录总共有多少个节点
for j in range(m):
    for i in range(n):
        p[i][j] = s + 1 if buffer[i][j] == 'E' else 0
        if p[i][j]:
            s += 1

# map 数组存储拉普拉斯矩阵的紧凑形式
# 因为网格宽度最大为 6，所以非零元素只会出现在主对角线附近（带状矩阵）
# 这里的 map[i][6] 代表主对角线元素 L[i][i]
map = [[0] * 13 for _ in range(600030)]

# 构建拉普拉斯矩阵：L = D (度数矩阵) - A (邻接矩阵)
for i in range(n):
    for j in range(m):
        if p[i][j]:
            curr_id = p[i][j] - 1 # 转换为从 0 开始的索引
            # 检查四个方向（上下左右）是否有连接的节点
            for di, dj in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                ni, nj = i + di, j + dj
                if 0 <= ni < n and 0 <= nj < m and p[ni][nj]:
                    neighbor_id = p[ni][nj] - 1
                    # 矩阵树定理规定：
                    # 1. 主对角线元素 L[i][i] = 节点的度数
                    map[curr_id][6] += 1 
                    # 2. 非对角线元素 L[i][j] = -1 (如果 i, j 相连)
                    # 由于是带状存储，偏移量计算为 6 - curr_id + neighbor_id
                    map[curr_id][6 - curr_id + neighbor_id] -= 1

res = 1
# 高斯消元计算行列式（主行列式的一阶余子式）
# 循环范围为 s-1，因为生成树数量等于拉普拉斯矩阵任意 n-1 阶余子式的值
for i in range(s - 1):
    # 如果主对角线元素为 0（理论上本题图中不会出现，除非图不连通）
    # map[i][6] 是当前的主元
    res = (res * map[i][6]) % mod
    
    # 只需要消去当前行下方宽度为 n (即 6) 的行，这是带状矩阵的优化关键
    for j in range(1, min(6, s - 1 - i) + 1):
        if map[i + j][6 - j] != 0:
            # 计算消元倍数 scale = L[i+j][i] / L[i][i]
            # 这里使用了费马小定理求逆元：pow(a, mod-2, mod)
            scale = (map[i + j][6 - j] * pow(map[i][6], mod - 2, mod)) % mod
            
            # 更新下方的行，只处理带状范围内的元素
            for k in range(6, 13):
                # L[i+j][k-j] -= L[i][k] * scale
                map[i + j][k - j] = (map[i + j][k - j] - map[i][k] * scale) % mod

# 输出最终结果（行列式的值即为生成树个数）
print((res + mod) % mod)
```

### 核心知识点补充：

1. **矩阵树定理 (Matrix Tree Theorem)**:

   一个图的生成树数量等于其拉普拉斯矩阵 $L$ 的任何一个 $n-1$ 阶主余子式的行列式值。

2. **带状矩阵优化**:

   在普通的 $N \times N$ 矩阵中，高斯消元需要 $O(N^3)$。但本题中，节点 $i$ 只与 $i \pm 1$ 或 $i \pm n$ 的节点相邻。这意味着在矩阵中，非零元素只分布在主对角线左右各 $n$ 的范围内。代码通过 `range(6, 13)` 和 `min(6, ...)` 极大地减少了计算量。

3. **模逆元**:

   在模运算下无法直接进行除法，所以 `a / b % mod` 必须写成 `a * pow(b, mod - 2, mod) % mod`。

n = int(input())

mem = [float("inf") for i in range(n)] #记录最短路径，初始化为正无穷，对于下标i，mem[i]表示1-i的最短路

from collections import deque #双端队列

que = deque()
que.append((1%n,0))  #%n主要是处理最小输入n=0当然单独处理0也行，从1开始，操作次数为0
while que:
  k,t = que.popleft()    
  if mem[k]>t:#如果当前方案比记录的更短就更新，当然处理得当第一次到达的就是最短，处理不好只能过75%
    mem[k] = t
    if k%2: #奇数只能*2

## 细胞克隆排序

```py
import sys

# 增加递归深度以防万一，虽然这里用不到
sys.setrecursionlimit(2000000)

def solve():
    # 使用快读
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    k = int(input_data[1])
    # 将能量值转为整数处理，比字符串快且准确
    a = list(map(int, input_data[2:]))

    # 预处理：找到每一个元素后面第一个不等于它的元素
    # 这一步是为了处理连续相同数字的情况，例如 (2, 2, 2, 5)
    # 只有当 A[i] != A[i+1] 时，克隆 i 还是 i+1 才会产生字典序差异
    next_diff = [n] * n
    for i in range(n - 2, -1, -1):
        if a[i] == a[i+1]:
            next_diff[i] = next_diff[i+1]
        else:
            next_diff[i] = i + 1

    # 定义下标的排序规则
    def compare_idx(i):
        # 我们要比较的是：克隆第 i 个产生的序列
        # 决定性因素是 a[i] 和 a[next_diff[i]] 的关系
        # 如果 a[i] < a[next_diff[i]]，克隆越早，字典序越小
        # 如果 a[i] > a[next_diff[i]]，克隆越晚，字典序越小
        
        # 这里有一个精妙的技巧：
        # 我们不需要写复杂的 cmp 函数，只需给每个下标生成一个可排序的特征值
        # 但对于 N=10^6，直接排序下标列表，并传入 key 是最稳妥的
        pass

    # 核心逻辑：
    # 实际上，这是一个经典的“删掉一个数使剩下最小”的变体。
    # 对于分裂问题，若 A[i] < A[i+1]，则克隆 i 比克隆 i+1 更小。
    # 若 A[i] > A[i+1]，则克隆 i+1 比克隆 i 更小。
    
    indices = list(range(n))
    
    # 自定义比较逻辑：比较克隆 i 和克隆 j (假设 i < j)
    # 它们第一个不同的位置是 i+1，分别是 A[i] 和 A[i+1]
    # 如果 A[i] != A[i+1]，则 A[i] < A[i+1] 时 i 更小
    # 如果 A[i] == A[i+1]，则看下一个不同的位置
    
    from functools import cmp_to_key
    
    def compare(i, j):
        if i == j: return 0
        first, second = min(i, j), max(i, j)
        # 找到从 first 开始第一个发生变化的位置
        diff_pos = next_diff[first]
        if diff_pos > second:
            # 说明在 [first, second] 之间全是相等的数
            # 这种情况下克隆哪一个产生的序列都一模一样
            return 0
        
        # 比较分歧点
        val_at_diff = a[diff_pos]
        val_original = a[first]
        
        if val_original < val_at_diff:
            res = -1 # i < j 时，左边的更小
        else:
            res = 1
            
        return res if i < j else -res

    # 排序下标
    indices.sort(key=cmp_to_key(compare))
    
    # 找到第 K 小的克隆位置
    best_idx = indices[k-1]
    
    # 输出结果：在 best_idx 处多打印一个 A[best_idx]
    ans = a[:best_idx] + [a[best_idx]] + a[best_idx:]
    print(*(ans))

solve()
```
