```py
先序遍历（根 → 左 → 右）:A → B → D → E → C
中序遍历（左 → 根 → 右）:D → B → E → A → C

class Solution:
    def inorderTraversal(self, root: Optional[TreeNode]) -> List[int]:
        res=[]

            if not node:
                return
            dfs(node.left)
            res.append(node.val)
            dfs(node.right)
        dfs(root)
        return res：
```

## 建树：

```py
def build_tree(nums):
    if not nums:
        return None
    root = TreeNode(int(nums[0]))
    from collections import deque
    q = deque([root])
    idx = 1
    while q and idx < len(nums):
        node = q.popleft()

        if idx < len(nums) and nums[idx] != "null":
            node.left = TreeNode(int(nums[idx]))
            q.append(node.left)
        idx += 1

        if idx < len(nums) and nums[idx] != "null":
            node.right = TreeNode(int(nums[idx]))
            q.append(node.right)
        idx += 1
    return root
```

```
先让数组第一个数字作为根节点

导进来队列（先进先出）

单独把根节点作为单数组创建队列

然后记录下一个要处理的节点索引就是idx=1（从一开始）

然后就是如果这个队列不为空且这个节点还没完呢就持续把队列首个拿出来作为节点node，然后判断只要没越界而且不是null的话就是把当时的node的left和right挂上去

对于第一次循环 len（q）=0->1->0->1->2
```

```py
from typing import Optional


class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        self.prev = float('-inf')  # 记录上一个节点的值，初始为负无穷
#先定义初始化的前节点是负无穷
        def inorder(node):
            if not node:
                #节点为空的情况
                return True
            # 左
            if not inorder(node.left):#not true#叶子节点,一般都是True，什么情况是False呢下面再说
                return False
            #这里就是再说比如说某一个node的left（这颗树正常排序出来的数组）并没有大于前面的节点
            # 根：判断是否严格递增
            if node.val <= self.prev:#迭代到最后的根节点（差一点是叶子节点）
                return False
            self.prev = node.val  # 更新上一个值 更新根节点作为第一个节点
            # 右[1,2,3] [1 5,3,4,6]
            return inorder(node.right)
        return inorder(root)
```

```py
class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        # 定义树节点
        class TreeNode:
            def __init__(self, val=0, left=None, right=None):
                self.val = val
                self.left = left
                self.right = right
        from typing import Optional
        class Solution:
            def recoverTree(self, root: Optional[TreeNode]) -> None:
                """
                Do not return anything, modify root in-place instead.
                """
                self.first = None  # 第一个错误节点
                self.second = None  # 第二个错误节点
                self.prev = TreeNode(float('-inf'))  # 前一个节点
                # 中序遍历找错误节点
                def inorder(node):
                    if not node:
                        return
                    inorder(node.left)
                    # 找到逆序：前一个节点 > 当前节点
                    if self.prev.val > node.val:
                        if not self.first:
                            self.first = self.prev  # 第一个错误
                        self.second = node  # 第二个错误
                    self.prev = node
                    inorder(node.right)
                inorder(root)
                # 交换两个错误节点的值
                self.first.val, self.second.val = self.second.val, self.first.val
```

## 判断树是否完全

```py
def isSameTree(p: TreeNode, q: TreeNode) -> bool:
    # 情况1：都为空，相同
    if not p and not q:
        #这上面是叶子节点的判断空=空
        return True
    # 情况2：一个空一个不空，不同
    if not p or not q:
        return False
    # 情况3：值不同，直接不同；值相同，递归判断左右子树
    return p.val == q.val and isSameTree(p.left, q.left) and isSameTree(p.right, q.right)
```

```py
class Solution:
    def isSymmetric(self, root: TreeNode) -> bool:
        if not root:
            return True
        return self.isMirror(root.left, root.right)

    def isMirror(self, p, q):
        # 都空
        if not p and not q:
            return True
        # 一个空一个不空
        if not p or not q:
            return False
        # 值相等 + 交叉镜像
        return p.val == q.val \
            and self.isMirror(p.left, q.right) \
            and self.isMirror(p.right, q.left)
```

## 递归深度：

```py
class Solution:
    def maxDepth(self, root: Optional[TreeNode]) -> int:
        # 递归终止条件：空节点深度为 0
        if not root:
            return 0

        # 左子树深度
        left_depth = self.maxDepth(root.left)
        # 右子树深度
        right_depth = self.maxDepth(root.right)

        # 当前节点深度 = 1 + 左右最大深度
        return 1 + max(left_depth, right_depth)
```

## 判断是否是平衡树：

```py
class Solution:
    def isBalanced(self, root: Optional[TreeNode]) -> bool:
        # 求高度 + 顺便判断是否平衡
        def check(node):
            if not node:
                return 0
            left = check(node.left)
            right = check(node.right)

            # 只要一边不平衡，或高度差>1，就返回-1表示不平衡
            if left == -1 or right == -1 or abs(left - right) > 1:
                return -1
            #-1的意思就是子树本身不是平衡树，abs意思是根节点作为子树来判断是否是平衡树
            # 否则返回正常高度
            return 1 + max(left, right)
        #总递归算出每一颗树的深度

        return check(root) != -1
    #只要不满足上面描
```

```yp
class Solution:
    def minDepth(self, root: Optional[TreeNode]) -> int:
        if not root:
            return 0

        q = deque([root])
        depth = 0

        while q:
            depth += 1
            for _ in range(len(q)):
                node = q.popleft()
                # 遇到叶子，直接返回当前深度
                if not node.left and not node.right:
                    return depth
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)

        return depth
```

## 判断是否存在路径从根节点到叶子节点和一定（为targetSUM）：

```py
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def hasPathSum(self, root: Optional[TreeNode], targetSum: int) -> bool:
        if not root:
            return False
        if not root.left and not root.right:
            return root.val==targetSum
        left=self.hasPathSum(root.left,targetSum-root.val)
        right=self.hasPathSum(root.right,targetSum-root.val)
        return left or right
```

## 输出路径：

```py
class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        res = []
        def dfs(node, path, remain):
            if not node:
                return
            path.append(node.val)
            # 到达叶子节点且和满足
            if not node.left and not node.right:
                if remain == node.val:
                    res.append(path.copy())
            dfs(node.left, path, remain - node.val)
            dfs(node.right, path, remain - node.val)
            # 回溯
            path.pop()

        dfs(root, [], targetSum)
        return res
```

## 二叉树展开成单链表

```py

class Solution:
    def flatten(self, root: TreeNode) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        if not root:
            return

        # 递归展开左右子树
        self.flatten(root.left)
        self.flatten(root.right)

        # 保存右子树
        temp_right = root.right

        # 左子树移到右边，左子树置空
        root.right = root.left
        root.left = None

        # 找到现在右子树的最后一个节点，接上原来的右子树
        current = root
        while current.right:
            current = current.right
        current.right = temp_right
        return current
```
